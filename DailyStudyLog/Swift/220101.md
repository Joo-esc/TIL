// MARK: MODEL

struct MemoryGame <CardContent> where CardContent: Equatable {
    private(set) var cards: [Card]
    private(set) var prevIndex: Int? {
        get {
            // return part
            // if prevIndex has value return nill value
            // if is Selected length is 1 add 'nill' value
            return cards.indices.filter({index in cards[index].isSelected}).onAndOnly
           
        }
        set {cards.indices.forEach({index in cards[index].isSelected = (index == newValue)})}
    }

    
    mutating func chooseCard(card: Card) {
        var selectedIndex = card.id
        if !cards[selectedIndex].isMatched && !cards[selectedIndex].isSelected {
            if let willMatchedIndex = prevIndex {
                // prevIndex is not nill
                if cards[willMatchedIndex].content == cards[selectedIndex].content {
                    cards[willMatchedIndex].isMatched = true
                    cards[selectedIndex].isMatched = true
                }
             
            } else {
                prevIndex = selectedIndex
               
            }
        }
        cards[selectedIndex].isSelected = true
    }
    
    init(piarOfCardIndex: Int, cardsContent: (Int) -> CardContent) {
        // add pair of cards
        cards = []
        for index in 0 ..< piarOfCardIndex {
            let content = cardsContent(index)
            cards.append(Card(content: content, id: index * 2))
            cards.append(Card(content: content, id: index * 2 + 1))
        }
    }
    
    struct Card: Identifiable {
        var isSelected = false
        var isMatched = false
        let content: CardContent
        let id: Int
    }
}

extension Array {
    var onAndOnly: Element? {
        if self.count == 1 {
            return first
        } else {
            return nil
        }
    }
}

// MARK: VIEW MODEL

import SwiftUI

class EmojiMemoryGame: ObservableObject {
    typealias Card = MemoryGame<String>.Card
    static let emojis = ["😈", "😎", "💫", "👻", "👹", "🙉", "❤️", "💩", "👺", "💛", "💕", "😬", "😏", "😊"]
    @Published private var model: MemoryGame<String> = MemoryGame<String>(piarOfCardIndex: 4) { index in
     emojis[index]
    }
    
    var cards: [Card] {
         model.cards
    }
    
    // MARK: business Logic
    func chooseCard(card: Card) {
        model.chooseCard(card: card)
    }
}

import SwiftUI

let emojis = ["😈", "😎", "💫", "👻", "👹", "🙉", "❤️", "💩", "👺", "💛", "💕", "😬", "😏", "😊"]

struct EmojisMemoryGameView: View {
    @ObservedObject var game: EmojiMemoryGame
    var body: some View {
        VStack {
            LazyVGrid(columns: [GridItem(.flexible()), GridItem(.flexible()), GridItem(.flexible()), GridItem(.flexible())]) {
                ForEach(game.cards) { card in
                    CardView(card: card)
                        .aspectRatio(2/3, contentMode: .fit)
                        .onTapGesture {
                            game.chooseCard(card: card)
                        }
                }
            }
            Spacer()
        }
        .padding()
    }
}

struct CardView: View {
    let card: EmojiMemoryGame.Card
    var body: some View {
        let shape = RoundedRectangle(cornerRadius: 16.0)
        VStack {
            ZStack {
                if card.isSelected {
                    shape.foregroundColor(Color.white)
                    shape.strokeBorder(lineWidth: 3).foregroundColor(Color.red)
                    Text(card.content)
                } else if card.isMatched {
                    shape.foregroundColor(Color.white)
                    shape.strokeBorder(lineWidth: 3).foregroundColor(Color.red)
                    Text(card.content)
                }
                else {
                    shape.foregroundColor(Color.red)
                }
            }
        }
    }
}

}
