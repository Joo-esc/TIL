import SwiftUI

struct ContentView: View {
    var emojis = ["🐰", "👀", "🧠", "💅", "🐰", "👀", "🧠", "💅", "🐰", "👀", "🧠", "💅" ]
    @ State var emojiCount: Int = 4
    var body: some View {
        VStack {
            HStack {
                // Need Idenfier , To match Up the View
                ForEach(emojis[0..<emojiCount], id: \.self) { emoji in
                    CardView(content: emoji)
                }
            }
            Spacer()
            HStack {
                add
                Spacer()
                remvoe
            }
            .font(.largeTitle)
            .padding(.horizontal
            )
        }
        .padding(.horizontal)
        .foregroundColor(.red)
    }
    /* Refacotring  */
    var add :some View {
        Button {
            if emojiCount < emojis.count {
                emojiCount += 1
            }
        } label: {
            VStack {
                Image(systemName: "minus.circle")
            }
        }
    }
    var remvoe :some View {
        Button {
            if emojiCount > 1 {
                emojiCount -= 1
            }
        } label: {
            VStack {
                Image(systemName: "plus.circle")
            }
        }
    }
}

struct CardView: View {
    var content: String
    /* Variable Must Have A Value (It means Functional Should get Paramter if It has Value)
     + If There is No Initial Value */
    // Giving Initial Value
    // It's Pointer. Changing Temp State, Not the real one
    @ State var isFaceUp: Bool = true
    var body: some View {
        return ZStack {
            // Local Valuable
            let shape = RoundedRectangle(cornerRadius: 20)
            if isFaceUp {
                shape.fill().foregroundColor(.white)
                shape.stroke(lineWidth: 3)
                Text(content).font(.largeTitle)
            } else {
                shape.fill()
            }
        }.onTapGesture {
            isFaceUp = !isFaceUp
            /* Every View int not Immutable  */
        }
    }
}

struct ContentView_Previews: PreviewProvider {
    static var previews: some View {
        ContentView()
            .preferredColorScheme(.dark)
        ContentView()
            .preferredColorScheme(.light)
    }
}
